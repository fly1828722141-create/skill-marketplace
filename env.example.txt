# ===========================================
# Skill Marketplace 环境变量配置模板
# ===========================================
# 复制此文件为 .env.local 并填入实际值
# 命令：cp env.example.txt .env.local

# ===========================================
# 数据库配置 (PostgreSQL)
# ===========================================
# FaaS 平台会自动注入 DATABASE_URL
# 本地开发示例：postgresql://user:password@localhost:5432/skill_marketplace
DATABASE_URL=your_database_url_here

# ===========================================
# NextAuth.js / Google 登录配置
# ===========================================
# Google OAuth 2.0
GOOGLE_CLIENT_ID=your_google_client_id
GOOGLE_CLIENT_SECRET=your_google_client_secret

# NextAuth Secret（用于加密 token，生产环境必须设置）
NEXTAUTH_SECRET=your_nextauth_secret_generate_by_openssl_rand_base64_32
NEXTAUTH_URL=http://localhost:3000

# ===========================================
# 对象存储配置（二选一）
# ===========================================
# STORAGE_PROVIDER 可选：oss / r2
# 不填时：优先尝试 r2，其次 oss，均未配置则回退数据库内置存储
STORAGE_PROVIDER=oss

# 方案 A：阿里云 OSS
OSS_REGION=oss-cn-hangzhou
OSS_BUCKET=skill-marketplace
OSS_ACCESS_KEY_ID=your_oss_access_key_id
OSS_ACCESS_KEY_SECRET=your_oss_access_key_secret
OSS_ENDPOINT=https://oss-cn-hangzhou.aliyuncs.com

# 方案 B：Cloudflare R2（S3 兼容）
# 注意：R2 模式下建议配置公开域名，保证评论图片与下载链接可直接访问
R2_ACCOUNT_ID=your_cloudflare_account_id
R2_BUCKET=your_r2_bucket_name
R2_ACCESS_KEY_ID=your_r2_access_key_id
R2_SECRET_ACCESS_KEY=your_r2_secret_access_key
R2_PUBLIC_BASE_URL=https://pub-xxxx.r2.dev

# ===========================================
# GitHub 压缩包发布（方案 B）
# ===========================================
# 作用：上传页“GitHub 压缩包发布”模式会把压缩包自动提交到指定仓库
# 必填：GITHUB_TOKEN / GITHUB_SKILL_OWNER / GITHUB_SKILL_REPO
GITHUB_TOKEN=your_github_personal_access_token
GITHUB_SKILL_OWNER=your_github_owner
GITHUB_SKILL_REPO=your_github_repo
GITHUB_SKILL_BRANCH=main
GITHUB_SKILL_BASE_PATH=skills

# ===========================================
# 应用配置
# ===========================================
# 最大上传文件大小（MB）
MAX_FILE_SIZE_MB=50

# 允许的文件类型
ALLOWED_FILE_TYPES=.zip,.tar.gz,.rar,.7z

# ===========================================
# 数据埋点与看板（可选）
# ===========================================
# 若配置 PostHog，会在本地 event_logs 入库的同时同步到 PostHog
POSTHOG_HOST=https://app.posthog.com
POSTHOG_PROJECT_API_KEY=your_posthog_project_api_key

# FaaS 平台环境变量（自动注入）
